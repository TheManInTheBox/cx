137  pFormalId;

140  TType *pFormalType = pFormalId->pType;
141  GetToken();
142 
143  /* Reference parameter: ExecuteVariable will leave the actual
144  * parameter's address on top of the stack. */
145  if (pFormalId->defn.how == dcReference) {
146  ExecuteVariable(pNode, true);
147  pFormalId->runstackItem = TOS();
148  GetToken();
149  }//--Value parameter
150  else {
151  TType *pActualType = ExecuteExpression();
152 
153  if ((pFormalType == pFloatType) &&
154  (pActualType->Base() == pIntegerType)) {

157  //--Convert integer value to real.
158  Push(float(Pop()->__int));
159  pFormalId->runstackItem = TOS();
160  } else if (!pFormalType->IsScalar()) {
161

163  //--Make a copy of the actual parameter's value.
164  void *addr = new char[pFormalType->size];
165  memcpy(addr, Pop()->__addr, pFormalType->size);
166  Push(addr);
167  pFormalId->runstackItem = TOS();
168  } else {
169

171  //--formal parameter.
