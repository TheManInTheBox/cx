46  TIcode *pIcode) {

48  TStackItem *pNewFrameBase = tos + 1; /* point to item just above
49  * current TOS item */

53  //--Compute the static link.
54  if (newLevel == oldLevel + 1) {
55

57  *--Push address of caller's stack frame.*/

62  --Push address of common parent's stack frame.*/
63  Push(pHeader->staticLink.__addr);

177 TStackItem *TRuntimeStack::GetValueAddress(const TSymtabNode *pId) {
178  bool functionFlag = pId->defn.how == dcFunction; // true if function
179  // else false
180  TFrameHeader *pHeader = (TFrameHeader *) pFrameBase;
181 
182  /*--Compute the difference between the current nesting level
183  *--and the level of the variable or parameter. Treat a function
184  *--value as if it were a local variable of the function. (Local
185  *--variables are one level higher than the function name.)*/
186  int delta = currentNestingLevel - pId->level;
187  if (functionFlag) --delta;
