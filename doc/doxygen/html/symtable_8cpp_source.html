222 cx_symtab_node *cx_symtab::search(const char *p_string) const {
223  cx_symtab_node *p_node = root__; // ptr to symbol table node
224  int comp;
225 
226  // Loop to search the table.
227  while (p_node) {
228  comp = strcmp(p_string, p_node->p_string); // compare names
229  if (comp == 0) break; // found!
230 
231  // Not yet found: next__ search left__ or right__ subtree.
232  p_node = comp < 0 ? p_node->left__ : p_node->right__;
