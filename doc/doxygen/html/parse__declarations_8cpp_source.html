173 void cx_parser::parse_constant(cx_symtab_node *p_const_id) {
174  cx_token_code sign = tc_dummy;
175 
176  if (token_in(token, tokenlist_unary_ops)) {
177  if (token == tc_minus) sign = tc_minus;
178  get_token_append();
179  }
180 
181  switch (token) {
182  case tc_number:
183  if ((p_token->type() == ty_integer) && (p_const_id->p_type == p_integer_type)) {
184  p_const_id->defn.constant.value.int__ = sign == tc_minus ?
185  -p_token->value().int__ : p_token->value().int__;
186  } else if ((p_token->type() == ty_real) &&
187  (((p_const_id->p_type == p_float_type)))) {
188 
189  if (p_const_id->p_type == p_float_type) {
190  p_const_id->defn.constant.value.float__ = sign == tc_minus ?
191  -p_token->value().float__ : p_token->value().float__;
192  } else {
193  p_const_id->defn.constant.value.double__ = sign == tc_minus ?
194  -p_token->value().float__ : p_token->value().float__;
195  }
196  }
197 
198  get_token_append();
199  break;
200 
201  case tc_identifier:
202  parse_identifier_constant(p_const_id, sign);
203  break;
204  case tc_char:
205  case tc_string:
